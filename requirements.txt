          var KEY = 'qdi_chunk_reload';
          var MAX = 2;

          function getCount() {
            try { return parseInt(sessionStorage.getItem(KEY) || '0', 10); } catch(e) { return 0; }
          }
          function setCount(n) {
            try { sessionStorage.setItem(KEY, String(n)); } catch(e) {}
          }
          function resetCount() {
            try { sessionStorage.removeItem(KEY); } catch(e) {}
          }

          window.addEventListener('error', function(ev) {
            var t = ev && ev.target;
            if (t && (t.tagName === 'SCRIPT' || t.tagName === 'LINK')) {
              var src = t.src || t.href || '';
              if (src.indexOf('/_next/static/') !== -1) {
                var c = getCount();
                if (c < MAX) {
                  setCount(c + 1);
                  window.location.reload();
                }
              }
            }
          }, true);

          window.addEventListener('load', function() {
            setTimeout(function() { resetCount(); }, 5000);
          });
        } catch(e) {}
      })();
